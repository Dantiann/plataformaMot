--Procedimiento almacenado que permite eliminar un registro de la tabla tblActividades
CREATE PROCEDURE spDeleteActivity
	@IdActividad int
AS
BEGIN
	DELETE FROM tblActividades
	WHERE idActividad = @IdActividad
END;
