--Procedimiento almacenado que permite eliminar un registro de la tabla tblRegistroAsistencias
CREATE PROCEDURE spDeleteAssitence
	@IdAsistencia int
AS
BEGIN
	DELETE tblRegistroAsistencias
	WHERE idAsistencia=@IdAsistencia;
END;
