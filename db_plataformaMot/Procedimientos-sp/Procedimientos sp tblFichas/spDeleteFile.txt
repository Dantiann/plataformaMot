--Procedimiento almacenado que permite eliminar un registro de la tabla tblFichas
CREATE PROCEDURE spDeleteFile
	@FichaNumero varchar(20)
AS
BEGIN
	DELETE FROM tblFichas
	WHERE fichaNumero=@FichaNumero;
END;
